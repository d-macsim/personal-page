**Step 2 -- Wrap ALL script logic in astro:page-load event:**

The entire `<script>` block in BaseLayout.astro currently runs once. With ClientRouter, it needs to re-run after each client-side navigation. Replace the entire script block content. Wrap all existing logic inside a function called `initPage`, then register it with the astro:page-load event:

```javascript
function setupNavObserver() {
  const sections = document.querySelectorAll('section[id]');
  const navLinks = document.querySelectorAll('[data-nav-link]');

  const observer = new IntersectionObserver(
    (entries) => {
      entries.forEach((entry) => {
        if (entry.isIntersecting) {
          navLinks.forEach((navLink) => {
            const href = navLink.getAttribute('href');
            if (href === `#${entry.target.id}`) {
              navLink.setAttribute('data-active', 'true');
            } else {
              navLink.removeAttribute('data-active');
            }
          });
        }
      });
    },
    { rootMargin: '-40% 0px -55% 0px', threshold: 0 }
  );

  sections.forEach((section) => observer.observe(section));
}

function initPage() {
  // Hamburger menu toggle
  const hamburgerBtn = document.getElementById('hamburger-btn');
  const mobileMenu = document.getElementById('mobile-menu');
  const hamburgerIcon = document.getElementById('hamburger-icon');
  const closeIcon = document.getElementById('close-icon');

  hamburgerBtn?.addEventListener('click', () => {
    const isOpen = !mobileMenu?.classList.contains('hidden');
    mobileMenu?.classList.toggle('hidden', isOpen);
    hamburgerIcon?.classList.toggle('hidden', !isOpen);
    closeIcon?.classList.toggle('hidden', isOpen);
    hamburgerBtn.setAttribute('aria-expanded', String(!isOpen));
  });

  // Close mobile menu when a link is clicked
  const mobileLinks = document.querySelectorAll('#mobile-menu a');
  mobileLinks.forEach((link) => {
    link.addEventListener('click', () => {
      mobileMenu?.classList.add('hidden');
      hamburgerIcon?.classList.remove('hidden');
      closeIcon?.classList.add('hidden');
      hamburgerBtn?.setAttribute('aria-expanded', 'false');
    });
  });

  // Smooth scroll for hash anchors only
  const allNavLinks = document.querySelectorAll('[data-nav-link]');
  allNavLinks.forEach((link) => {
    link.addEventListener('click', (e) => {
      const href = link.getAttribute('href');
      if (!href || !href.startsWith('#')) return;
      e.preventDefault();
      const target = document.querySelector(href);
      target?.scrollIntoView({ behavior: 'smooth' });
    });
  });

  // Mark /now link active on /now route
  if (window.location.pathname === '/now' || window.location.pathname === '/now/') {
    const nowLinks = document.querySelectorAll('[data-nav-link][href="/now"]');
    nowLinks.forEach((link) => link.setAttribute('data-active', 'true'));
  }

  setupNavObserver();
}

// astro:page-load fires on initial load AND after each ClientRouter navigation
document.addEventListener("astro:page-load", initPage);
```

Note: Do NOT call `initPage()` directly in addition to the event listener. The `astro:page-load` event fires on initial page load as well, so a direct call would cause double-binding. Only use the event listener.
